Basic usage

$(function($){
    $("#date").mask("99/99/9999");
    $("#phone").mask("(999) 999-9999");
    $("#tin").mask("99-9999999");
    $("#ssn").mask("999-99-9999");
});

Custom placeholder

$(function($){
    $("#product").mask("99/99/9999",{placeholder:" "});   
});

Placeholder defenition

$(function($){
   $.mask.addPlaceholder('~',"[+-]");
   $("#eyescript").mask("~9.99 ~9.99 999");
});
