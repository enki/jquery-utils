Overwrite defaults

$(document).ready(function(){
    
    $('#demo-2').cycle([
        'images/cisoun-tux-kenny-2198.png',
        'images/mitch91-rayon-tux-2579.png',
        'images/sosoe222-smart-tux-3440.png'
    ], {
        interval: 10000, // default is 5000
        onBefore: function(element, image) { // default is false
            $(element).next().next().append($('
  • Loading: '+ image + '
    • '));
        },
        onAfter: function(element, image) { // default is false
            $(element).next().html('Loaded: '+ image +'');
        },
        callback: function(element, image){ // default is generic image rotation script
            if ($.browser.mozilla) window.scrollTo(window.pageXOffset, window.pageYOffset); // scrollfix
            $(element)
                .fadeOut()
                .queue(function(){
                    $(this).css({backgroundImage:'url("'+ image +'")'});
                    $(this).dequeue();
                    $(this).fadeIn('slow');
                    $(this).dequeue(); });
        }
    });
});
