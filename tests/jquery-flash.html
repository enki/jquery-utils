Documentation

Basic usage

$('#hello').flash({
    src: 'hello.swf',
    width: 320,
    height: 240
});

Custom replacement

$('.custom').flash(null, null, function(htmlOptions){
    // do stuff
});

Overwriting defaults

$.fn.flash.replace = function(htmlOptions) {
    // always do stuff
};

Flash detection

$('#hello').flash(
    { src: 'hello.swf' }, 
    { version: '6.0.65' }
);

Express install

$('#hello').flash(
    { src: 'hello.swf' }, 
    { expressInstall: true }
);

Graceful degradation

$('#hello').flash(
    { src: 'hello.swf' }, 
    { update: false }
);
