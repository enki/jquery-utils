Usage

$(document).ready(function(){
    $('div.ah-demo:not(first:)').hide();
    $.anchorHandler
     .add(/#demo\-.*/, function(){
          console.log(this.hash);
         $('div.ah-demo')
             .hide('fast')
             .filter('#'+ this.hash.replace('#',''))
             .slideDown('slow');
         return false; }, true);
});

// html

<p>
    <a href="#demo-1">demo 1</a>
    <a href="#demo-2">demo 2</a>
    <a href="#demo-3">demo 3</a>
</p>
<div class="demos">
    <div id="demo-1" class="ah-demo" style="background:#ccf;">demo 1</div>
    <div id="demo-2" class="ah-demo" style="background:#efc;">demo 2</div>
    <div id="demo-3" class="ah-demo" style="background:#c9f;">demo 3</div>
</div>
