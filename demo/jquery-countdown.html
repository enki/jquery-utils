Custom format

$('#countdown-1').countdown({
    date: new Date(2010, 0, 1, 0, 0, 0),
    msgNow: 'Happy new year!', 
    msgFormat: '%d [day|days], %h [hour|hours], %m [minute|minutes], %s [second|seconds]'
});

Date modifiers

$('#countdown-2').countdown({
    modifiers: ['+1y', '+30d', '+6h']
});
